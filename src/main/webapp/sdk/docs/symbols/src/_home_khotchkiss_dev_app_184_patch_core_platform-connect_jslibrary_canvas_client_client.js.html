 15         function startsWithHttp(u, d) {
 16             return  $$.isNil(u) ? u : (u.substring(0, 4) === "http") ? u : d;
 17         }

 19         // returns the url of the Parent Window
 20         function getTargetOrigin(to) {
 21             var h;
 22             if (to === "*") {return to;}
 23             if (!$$.isNil(to)) {
 24                 h = $$.stripUrl(to);
 25                 purl = startsWithHttp(h, purl);
 26                 if (purl) {return purl;}
 27             }
 28             // This relies on the parent passing it in. This may not be there as the client can do a redirect.
 29             h = $$.document().location.hash;
 30             if (h) {
 31                 h = decodeURIComponent(h.replace(/^#/, ''));
 32                 purl = startsWithHttp(h, purl);
 33             }
 34             return purl;
 35         }

 55             function postit(clientscb, message) {
 56                 var wrapped, to, c;
 57 
 58                 // need to keep a mapping from request to callback, otherwise
 59                 // wrong callbacks get called. Unfortunately, this is the only
 60                 // way to handle this as postMessage acts more like topic/queue.
 61                 // limit the sequencers to 100 avoid out of memory errors
 62                 seq = (seq > 100) ? 0 : seq + 1;
 63                 cbs[seq] = clientscb;
 64                 wrapped = {seq : seq, src : "client", clientVersion : cversion, parentVersion: pversion, body : message};
 65 
 66                 c  = message && message.config && message.config.client;
 67                 to = getTargetOrigin($$.isNil(c) ? null : c.targetOrigin);
 68                 if ($$.isNil(to)) {
 69                     throw "ERROR: targetOrigin was not supplied and was not found on the hash tag, this can result from a redirect or link to another page.";
 70                 }
 71                 $$.xd.post(wrapped, to, parent);
 72             }

 74             function validateClient(client, cb) {
 75                 var msg;
 76 
 77                 client = client || $$.oauth && $$.oauth.client();
 78 
 79                 if ($$.isNil(client) || $$.isNil(client.oauthToken)) {
 80                     msg = {status : 401, statusText : "Unauthorized" , parentVersion : pversion, payload : "client or client.oauthToken not supplied"};
 81                 }
 82                 if ($$.isNil(client.instanceId) || $$.isNil(client.targetOrigin)) {
 83                     msg = {status : 400, statusText : "Bad Request" , parentVersion : pversion, payload : "client.instanceId or client.targetOrigin not supplied"};
 84                 }
 85                 if (!$$.isNil(msg)) {
 86                     if ($$.isFunction(cb)) {
 87                         cb(msg);
 88                         return false;
 89                     }
 90                     else {
 91                         throw msg;
 92                     }
 93                 }
 94                 return true;
 95             }

308                      * @description Performs a cross-domain, asynchronous HTTP request.

319                      * @config {Boolean} [async=true] Asynchronous: only <code>true</code> is supported.

329                      * Sfdc.canvas.client.ajax(url,

331                      *     method: 'POST',
332                      *     contentType: "application/json",
333                      *     data: JSON.stringify(body),

335                      *     if (201 === data.status) {
336                      *          alert("Success"
337                      *          }
338                      *     }
339                      *   });
340                      * @example

342                      * // Paste the signed request string into a JavaScript object for easy access.
343                      * var sr = JSON.parse('<%=signedRequestJson%>');
344                      * // Reference the Chatter user's URL from Context.Links object.
345                      * var chatterUsersUrl = sr.context.links.chatterUsersUrl;
346                      *
347                      * // Make an XHR call back to Salesforce through the supplied browser proxy.
348                      * Sfdc.canvas.client.ajax(chatterUsersUrl,
349                      *   {client : sr.client,
350                      *   success : function(data){
351                      *   // Make sure the status code is OK.
352                      *   if (data.status === 200) {

354                      *     alert("Got back "  + data.payload.users.length +

357                      * })};

360

363                         if (!url) {
364                             throw "PRECONDITION ERROR: url required with AJAX call";
365                         }
366                         if (!settings || !$$.isFunction(settings.success)) {
367                             throw "PRECONDITION ERROR: function: 'settings.success' missing.";
368                         }
369                         if (! validateClient(settings.client, settings.success)) {
370                             return;
371                         }

373                         ccb = settings.success;
374                         defaults = {
375                             method: 'GET',
376                             async: true,
377                             contentType: "application/json",
378                             headers: {"Authorization" : "OAuth "  + settings.client.oauthToken,

380                             data: null
381                         };

394                         }

450                      * @description Returns the current size of the iFrame.
451                      * @return {Object}<br>
452                      * <code>heights.contentHeight</code>: the height of the virtual iFrame, all content, not just visible content.<br>
453                      * <code>heights.pageHeight</code>: the height of the visible iFrame in the browser.<br>
454                      * <code>heights.scrollTop</code>: the position of the scroll bar measured from the top.<br>
455                      * <code>widths.contentWidth</code>: the width of the virtual iFrame, all content, not just visible content.<br>
456                      * <code>widths.pageWidth</code>: the width of the visible iFrame in the browser.<br>
457                      * <code>widths.scrollLeft</code>: the position of the scroll bar measured from the left.

459                      * //get the size of the iFrame and print out each component.
460                      * var sizes = Sfdc.canvas.client.size();

480                     /**
481                      * @public
482                      * @name Sfdc.canvas.client#resize
483                      * @function
484                      * @description Informs the parent window to resize the canvas iFrame. If no parameters are specified,
485                      * the parent window attempts to determine the height of the canvas app based on the
486                      * content and then sets the iFrame width and height accordingly. To explicitly set the dimensions,
487                      * pass in an object with height and/or width properties.
488                      * @param {Client} client The object from the signed request
489                      * @param {size} size The optional height and width information
490                      * @example
491                      * //Automatically determine the size
492                      * Sfdc.canvas(function() {
493                      *     sr = JSON.parse('<%=signedRequestJson%>');
494                      *     Sfdc.canvas.client.resize(sr.client);
495                      * });
496                      *
497                      * @example
498                      * //Set the height and width explicitly
499                      * Sfdc.canvas(function() {
500                      *     sr = JSON.parse('<%=signedRequestJson%>');
501                      *     Sfdc.canvas.client.resize(sr.client, {height : "1000px", width : "900px"});
502                      * });
503                      *
504                      * @example
505                      * //Set only the height
506                      * Sfdc.canvas(function() {
507                      *     sr = JSON.parse('<%=signedRequestJson%>');
508                      *     Sfdc.canvas.client.resize(sr.client, {height : "1000px"});
509                      * });
510                      *
511                      */
512                     resize : function(client, size) {
513                         var sh, ch, sw, cw, s = {height : "", width : ""},
514                             docElement = $$.document().documentElement;
515 
516                         // If the size was not supplied, adjust window
517                         if ($$.isNil(size)) {
518                             sh = docElement.scrollHeight;
519                             ch = docElement.clientHeight;
520                             if (ch !== sh) {
521                                 s.height = sh + "px";
522                             }
523                             sw = docElement.scrollWidth;
524                             cw = docElement.clientWidth;
525                             if (sw !== cw) {
526                                 s.width = sw + "px";
527                             }
528                         }
529                         else {
530                             if (!$$.isNil(size.height)) {
531                                 s.height = size.height;
532                             }
533                             if (!$$.isNil(size.width)) {
534                                 s.width = size.width;
535                             }
536                         }
537                         if (!$$.isNil(s.height) || !$$.isNil(s.width)) {
538                             postit(null, {type : "resize", config : {client : client}, size : s});
539                         }

566                      * });
